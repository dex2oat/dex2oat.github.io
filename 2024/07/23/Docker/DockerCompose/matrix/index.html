matrix

Matrix 是一个开源、可交互、去中心化的实时通信服务框架。使用 Matrix 可以搭建安全的通信服务器，配合支持 Matrix 的客户端可以实现个人、团队间的实时聊天交互，对Matrix Dendrite搭建聊天服务器的过程感兴趣的朋友一起看看吧

部署方式

docker-compose

version: "3"
services:
  db:
    restart: always
    image: postgres:latest
    shm_size: 1024mb
    container_name: synapse-postgres
    networks:
      - dbs_network
    healthcheck:
      test: ["CMD", "pg_isready", "-U", "postgres"]
    volumes:
      - ./postgres:/var/lib/postgresql/data
    environment:
      - "PUID=1030"
      - "PGID=100"
      - "TZ:=Asia/Shanghai"

  redis:
    restart: always
    image: redis:latest
    container_name: synapse-redis
    networks:
      - dbs_network
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
    volumes:
      - ./redis:/data
    environment:
      - "PUID=1030"
      - "PGID=100"
      - "TZ:=Asia/Shanghai"

  synapse:
    image: matrixdotorg/synapse:latest
    container_name: matrix-synapse
    ports:
      - 38008:8008
    depends_on:
      - db
      - redis
    volumes:
      - ./data:/data
    restart: always
    networks:
      - synapse_network
      - dbs_network
    environment:
      - "PUID=1030"
      - "PGID=100"
      - "TZ:=Asia/Shanghai"
networks:
  synapse_network:
  dbs_network:
    internal: true
